📋 전체 알림 시스템 흐름 (이해 필수)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

[기사 작성 / 댓글 작성]
     ↓
sendNotification() 호출
     ↓
Firebase DB에 notifications/{uid}/{notifId} 저장
  (pushed: false 로 저장됨)
     ↓
① 앱이 열려있으면 → onMessage() 핸들러 → 토스트 팝업 표시
② 앱이 닫혀있으면 → GitHub Actions (5분마다 실행)
                    → send-notifications.js
                    → pushed=false 인 알림 검색
                    → FCM 서버로 전송
                    → 기기에 푸시 알림 도착
                    → pushed: true 로 업데이트

🧪 STEP 6. 내 알림 목록 전체 조회
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

// 내 알림 전체 조회 (최신순)
db.ref("notifications/" + getUserId())
    .orderByChild("timestamp")
    .limitToLast(20)
    .once("value")
    .then(snap => {
        const all = snap.val() || {};
        const list = Object.entries(all)
            .map(([id, n]) => ({
                id,
                title: n.title,
                type: n.type,
                pushed: n.pushed,     // true면 FCM 전송 완료
                read: n.read,
                timestamp: new Date(n.timestamp).toLocaleString('ko-KR')
            }))
            .sort((a, b) => b.timestamp - a.timestamp);
        
        console.table(list);
        console.log("총 알림 수:", Object.keys(all).length);
        
        const notPushed = Object.values(all).filter(n => !n.pushed);
        console.log("FCM 미전송 알림 수:", notPushed.length, "(pushed=false)");
    });

🧪 STEP 8. 특정 알림의 pushed 상태 수동 초기화
       (이미 전송된 알림을 다시 테스트하고 싶을 때)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

// 특정 알림 ID의 pushed 를 false 로 리셋 (재전송 대기 상태로)
var notifIdToReset = "여기에알림ID입력";
db.ref("notifications/" + getUserId() + "/" + notifIdToReset).update({
    pushed: false,
    read: false
}).then(() => {
    console.log("✅ 알림 리셋 완료 - 다음 GitHub Actions 실행 시 재전송됨");
});

// 내 모든 알림의 pushed 를 false 로 리셋 (전체 재전송 테스트)
db.ref("notifications/" + getUserId()).once("value").then(snap => {
    const all = snap.val() || {};
    const updates = {};
    Object.keys(all).forEach(id => {
        updates[id + "/pushed"] = false;
        updates[id + "/read"] = false;
    });
    return db.ref("notifications/" + getUserId()).update(updates);
}).then(() => {
    console.log("✅ 모든 알림 pushed=false 리셋 완료");
});

📊 알림 시스템 상태 한번에 진단 (복사해서 한 번에 실행)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

(async () => {
    console.log("============ 알림 시스템 진단 ============");
    
    // 1. 로그인
    console.log("1. 로그인:", isLoggedIn() ? "✅ " + getUserEmail() : "❌ 로그인 필요");
    
    // 2. 브라우저 권한
    console.log("2. 브라우저 알림권한:", Notification.permission === "granted" ? "✅ 허용" : "❌ " + Notification.permission);
    
    // 3. Service Worker
    const regs = await navigator.serviceWorker.getRegistrations();
    console.log("3. Service Worker:", regs.length > 0 ? "✅ " + regs.length + "개 등록됨" : "❌ 미등록");
    
    // 4. Firebase Messaging
    console.log("4. Firebase Messaging:", window.messaging ? "✅ 초기화됨" : "❌ 초기화 안됨");
    
    // 5. FCM 토큰
    if (isLoggedIn()) {
        const snap = await db.ref("users/" + getUserId() + "/fcmTokens").once("value");
        const tokens = snap.val() || {};
        const count = Object.keys(tokens).length;
        console.log("5. FCM 토큰:", count > 0 ? "✅ " + count + "개" : "❌ 없음 (registerFCMToken() 실행 필요)");
        
        // 6. 알림 설정
        const userSnap = await db.ref("users/" + getUserId()).once("value");
        const userData = userSnap.val() || {};
        const types = userData.notificationTypes || {};
        console.log("6. 알림 ON/OFF:", userData.notificationsEnabled !== false ? "✅ ON" : "❌ OFF");
        console.log("   - 기사 알림:", types.article !== false ? "✅ ON" : "❌ OFF");
        console.log("   - 댓글 알림:", types.comment !== false ? "✅ ON" : "❌ OFF");
        
        // 7. 대기중인 알림
        const notifSnap = await db.ref("notifications/" + getUserId())
            .orderByChild("pushed").equalTo(false).once("value");
        const pending = notifSnap.val() || {};
        const pendingCount = Object.keys(pending).length;
        console.log("7. FCM 전송 대기 알림:", pendingCount > 0 ? "⏳ " + pendingCount + "개 대기중" : "✅ 없음");
    }
    
    console.log("==========================================");
    console.log("❓ 문제 있을 경우:");
    console.log("  권한 거부 → 브라우저 주소창 자물쇠 클릭 → 알림 허용");
    console.log("  토큰 없음 → registerFCMToken() 실행");
    console.log("  대기 알림 없음 → STEP 5 명령어로 테스트 알림 생성");
})();
